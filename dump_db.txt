CREATE VIEW public.v_cooperativas_cards AS
 SELECT c.id_cooperativa AS id,
    c.nombre,
    c.ruc,
    COALESCE(NULLIF(btrim((c.telefono)::text), ''::text), NULLIF(btrim((c.telefono_movil)::text), ''::text), NULLIF(btrim((c.telefono_fijo_1)::text), ''::text), NULLIF(btrim((c.telefono_fijo_2)::text), ''::text)) AS telefono,

        CASE
            WHEN cs.activo THEN srv.nombre_servicio
            ELSE NULL::character varying
        END)::text, ', '::text ORDER BY srv.nombre_servicio) AS servicios_text,
    c.activa
   FROM (((((public.cooperativas c
     LEFT JOIN public.provincia p ON ((c.provincia_id = p.id)))
     LEFT JOIN public.canton t ON ((c.canton_id = t.id)))
     LEFT JOIN public.segmentos s ON ((c.id_segmento = s.id_segmento)))
     LEFT JOIN public.cooperativa_servicio cs ON ((c.id_cooperativa = cs.id_cooperativa)))
     LEFT JOIN public.servicios srv ON ((cs.id_servicio = srv.id_servicio)))
  GROUP BY c.id_cooperativa, c.nombre, c.ruc, c.telefono, c.telefono_movil, c.telefono_fijo_1, c.telefono_fijo_2, c.email, c.activa, p.nombre, t.nombre, c.id_segmento, s.nombre_segmento;
